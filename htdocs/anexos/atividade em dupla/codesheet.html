Como ler e exiber o nome com mvc:

1° passo database php: 

        

            

<?php

class Database
{
    // Variável que armazenará a conexão
    private $conn;

    // Método que cria e retorna a conexão
    public function connect()
    {
        $this->conn = new PDO(
            "mysql:host=localhost;dbname=teste",
            "root",
            ""
        );

        return $this->conn;
    }
}
2° passo model php: 

        

            


<?php

require_once 'database.php';

class UserModel
{
    // Guarda a conexão com o banco
    private $db;

    // Executado quando o objeto é criado
    public function __construct()
    {
        $database = new Database();

        // Obtém a conexão do arquivo database.php
        $this->db = $database->connect();
    }

    // Busca um usuário pelo ID
    public function getUser($id)
    {
        // Consulta SQL
        $sql = "SELECT nome FROM usuarios WHERE id = ?";

        // Prepara a consulta
        $stmt = $this->db->prepare($sql);

        // Executa passando o ID
        $stmt->execute([$id]);

        // Retorna o resultado
        return $stmt->fetch(PDO::FETCH_ASSOC);
    }
}
3° passo controller php: 

        

            



<?php

require_once 'model.php';

class UserController
{
    public function showUser()
    {
        // Cria o Model
        $userModel = new UserModel();

        // Busca o usuário de ID 1
        $user = $userModel->getUser(1);

        // Envia os dados para a View
        require 'home.php';
    }
}
